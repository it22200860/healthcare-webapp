pipeline {
    agent any

    environment {
        DOCKERHUB_USER = 'ravindunana'
        DOCKERHUB_PASS = credentials('dockerhub-password') // You'll add this credential next
    }

    stages {
        stage('Checkout Code') {
            steps {
                git branch: 'main', url: 'https://github.com/it22200860/healthcare-webapp.git'
            }
        }

        stage('Build Docker Image') {
            steps {
                sh 'docker build -t healthcare-webapp .'
            }
        }

        stage('Tag & Push to DockerHub') {
            steps {
                sh '''
                docker tag healthcare-webapp ${DOCKERHUB_USER}/healthcare-webapp:latest
                echo $DOCKERHUB_PASS | docker login -u $DOCKERHUB_USER --password-stdin
                docker push ${DOCKERHUB_USER}/healthcare-webapp:latest
                '''
            }
        }

        stage('Deploy to EC2 via Ansible') {
            steps {
                sshagent(['jenkins-ssh-key']) {
                    sh 'ansible-playbook -i ansible/hosts ansible/deploy.yml --ssh-extra-args="-o StrictHostKeyChecking=no"'
                }
            }
        }
    }

    post {
        success {
            echo "Deployment completed successfully!"
        }
        failure {
            echo "Deployment failed. Check Jenkins logs."
        }
    }
}
